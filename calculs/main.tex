  \noindent Longueur des tiges : $l$\\
  Longueur : $X=xl$\\
  Largeur : $Y=yl$\\
  Hauteur : $Z=zl$


  \begin{align}
    \text{nbr}_{\vec{x}} &= x\times y \times z\label{eq:1}\\
    \text{nbr}_{\vec{y}} &= x\times y \times z\label{eq:2}\\
    \text{nbr}_{\vec{z}} &= x\times y \times z\label{eq:3}\\
    \text{nbr} &= \text{nbr}_{\vec{x}} + \text{nbr}_{\vec{y}} + \text{nbr}_{\vec{z}} = 3xyz
  \end{align}
  \eqref{eq:1}, \eqref{eq:2} et \eqref{eq:3} sont égaux par symétrie. De plus, il n'y a pas de collisions car chaque barre est sur un axe différent.

  Donc,
  \begin{equation}
    \text{nbr}(l, X, Y, Z) \approx 3\frac{XYZ}{l^3}
  \end{equation}
  avec une égalité lorsque $X$, $Y$ et $Z$ sont \enquote{divisibles} par $l$.

  De plus, la longueur totale de tube nécessaire pour un meuble est :
  \begin{equation}
    \mathcal{L}\left(l, X, Y, Z\right)=\text{nbr}(l, X, Y, Z)\times l \approx 3\frac{XYZ}{l^2}
  \end{equation}

  Soit $r_i$ le rayon intérieur des tubes et $r_e$ le rayon extérieur. Le volume d'un tube $v$ est :
  \begin{equation}
    v=\left( r_e^2\pi - r_i^2\pi \right)l = \pi l\left( r_e^2 - r_i^2 \right)
  \end{equation}

  Soit $\rho$ la masse volumique de l'acier inoxydable. Un tube pèse $v\rho$. Donc le poids total du meuble est :
  \begin{equation}
    \text{pds} = 3\frac{XYZ}{l^3}v\rho=3\frac{\pi\rho}{l^2}\left( r_e^2 - r_i^2 \right)XYZ
  \end{equation}

  Soit $L$ la longueur des tubes fournis et $\mathcal{L}_L$ le nombre de ces tubes :
  \begin{equation}
    \mathcal{L}_L = \frac{\text{nbr}\left( l, X, Y, Z \right)}{\left\lfloor \frac{L}{l} \right\rfloor}
  \end{equation}

  Le volume d'un long tube est :
  \begin{equation}
    v_L=\pi L\left( r_e^2 - r_i^2 \right)
  \end{equation}

  Ce qui nous donne un poids de commande de :
  \begin{equation}
    \text{pds}_L=\mathcal{L}_Lv_L\rho=3\frac{\pi\rho}{l^3\left\lfloor \frac{L}{l} \right\rfloor}\left( r_e^2 - r_i^2 \right)XYZ
  \end{equation}

  Ensuite,
  \begin{equation}
    \text{pds}_L \geq 3\frac{\pi\rho}{l^3\left(\frac{L}{l} + 1 \right)}\left( r_e^2 - r_i^2 \right)XYZ=3\frac{\pi\rho}{l^2\left(L + l \right)}\left( r_e^2 - r_i^2 \right)XYZ
  \end{equation}
  Donc (puisque tout est positif) :
  \begin{equation}
    l^2\left(L + l \right) \geq 3\frac{\pi\rho}{\text{pds}_L}\left( r_e^2 - r_i^2 \right)XYZ
  \end{equation}
